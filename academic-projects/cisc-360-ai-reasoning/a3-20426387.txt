% Q1: A person likes tomato if they like some food that contains tomato.
likes_tomato(Person) :-
    likes(Person, Food),
    contains(Food, tomato).

% Q2: A person's favorite ingredient is one that shows up in a food they like.
favorite_ingredient(Person, Ingredient) :-
    likes(Person, Food),
    contains(Food, Ingredient).

% Q3: Two different people are friends if they like the same food.
friends(Person1, Person2) :-
    Person1 \= Person2,
    likes(Person1, Food),
    likes(Person2, Food).

% Q4: Two different people are compatible if they both like the same food and enjoy the same activity.
compatible(Person1, Person2) :-
    Person1 \= Person2,
    likes(Person1, Food),
    likes(Person2, Food),
    enjoys(Person1, Activity),
    enjoys(Person2, Activity).

% Q6: Flight network and recursive travel rule.
flight(paris, london).
flight(london, berlin).
flight(berlin, rome).
flight(paris, madrid).
flight(madrid, rome).

travel(X, Y) :-
    flight(X, Y).
travel(X, Y) :-
    flight(X, Z),
    travel(Z, Y).
